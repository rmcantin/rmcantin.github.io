66 int main(int nargs, char *args[])
67 {
68  int n = 10; // Number of dimensions
69  clock_t start, end;
70  double diff,diff2;
71 
72  // Common configuration
73  // See parameters.h for the available options.
74  // Some parameters did not need to be changed for default, but we have done it for
75  // illustrative purpose.
76  bopt_params par = initialize_parameters_to_default();
77 
78  set_kernel(&par,"kSum(kSEISO,kConst)");
79  par.kernel.hp_mean[0] = 1.0;
80  par.kernel.hp_mean[1] = 1.0;
81  par.kernel.hp_std[0] = 1.0;
82  par.kernel.hp_std[1] = 1.0;
83  par.kernel.n_hp = 2;
84 
85  set_mean(&par,"mConst");
86  par.mean.coef_mean[0] = 1.0;
87  par.mean.coef_std[0] = 10.0;
88  par.mean.n_coef = 1;
89 
90  set_surrogate(&par,"sStudentTProcessJef");
91  par.noise = 1e-10;
92 
93  par.sc_type = SC_MAP;
94  par.l_type = L_EMPIRICAL;
95 
96  par.n_iterations = 100; // Number of iterations
97  par.random_seed = 0;
98  par.n_init_samples = 15;
99  par.n_iter_relearn = 0;
100 
101  /*******************************************/
102  std::cout << "Running C++ interface" << std::endl;
103 
104  ExampleQuadratic opt(n,par);
105  vectord result(n);
106 
107  // Run C++ interface
108  start = clock();
109  opt.optimize(result);
110  end = clock();
111  diff = (double)(end-start) / (double)CLOCKS_PER_SEC;
112 
113  /*******************************************/
114  std::cout << "Running C inferface" << std::endl;
115 
116  // Prepare C interface
117  double low[128], up[128], xmin[128], fmin[128];
118 
119  // Lower and upper bounds
120  for (int i = 0; i < n; ++i)
121  {
122  low[i] = 0.;
123  up[i] = 1.;
124  }
125 
126  // Run C interface
127  start = clock();
128  bayes_optimization(n,&testFunction,NULL,low,up,xmin,fmin,par);
129  end = clock();
130  diff2 = (double)(end-start) / (double)CLOCKS_PER_SEC;
131  /*******************************************/
132 
133 
134  // Results
135  std::cout << "Final result C++: " << result << std::endl;
136  std::cout << "Elapsed time in C++: " << diff << " seconds" << std::endl;
137 
138  std::cout << "Final result C: [" << n <<"](" << xmin[0];

141  std::cout << "," << xmin[i];
142  }
143  std::cout << ")" << std::endl;
