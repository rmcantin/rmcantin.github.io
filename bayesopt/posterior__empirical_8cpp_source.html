81  void EmpiricalBayes::setCriteria(randEngine& eng)
82  {
83  CriteriaFactory mCFactory;
84 
85  mCrit.reset(mCFactory.create(mParameters.crit_name,mGP.get()));
86  mCrit->setRandomEngine(eng);
87 
88  if (mCrit->nParameters() == mParameters.n_crit_params)
89  {
90  mCrit->setParameters(utils::array2vector(mParameters.crit_params,
91  mParameters.n_crit_params));
92  }
93  else // If the number of paramerters is different, use default.
94  {
95  if (mParameters.n_crit_params != 0)
96  {
97  FILE_LOG(logERROR) << "Expected " << mCrit->nParameters()
98  << " parameters. Got "
99  << mParameters.n_crit_params << " instead.";
100  }
101  FILE_LOG(logINFO) << "Usign default parameters for criteria.";
102  }
103  } // setCriteria
