167  double evaluateSample( const vectord& xin)
168  {
169  double y = 0.0;
170  for(size_t i=0; i<4; ++i)
171  {
172  double sum = 0.0;
173  for(size_t j=0;j<6; ++j)
174  {
175  double val = xin(j)-mP(i,j);
176  sum -= mA(i,j)*val*val;
177  }
178  y -= mC(i)*std::exp(sum);
179  }
180  return y;
181  };
