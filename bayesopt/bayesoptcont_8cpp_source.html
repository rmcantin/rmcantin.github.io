86  void ContinuousModel::findOptimal(vectord &xOpt)
87  {
88  double minf = cOptimizer->run(xOpt);
89 
90  //Let's try some local exploration like spearmint
91  randNFloat drawSample(mEngine,normalDist(0,0.001));
92  for(size_t ii = 0;ii<5; ++ii)
93  {
94  vectord pert = getPointAtMinimum();
95  for(size_t j=0; j<xOpt.size(); ++j)
96  {
97  pert(j) += drawSample();
98  }
99  try
100  {
101  double minf2 = cOptimizer->localTrialAround(pert);
102  if (minf2<minf)
103  {
104  minf = minf2;
105  FILE_LOG(logDEBUG) << "Local beats Global";
106  xOpt = pert;
107  }
108  }
109  catch(std::invalid_argument& e)
110  {
111  //We ignore this one
112  }
113  }
114  };
