96 int bayes_optimization(int nDim, eval_func f, void* f_data,
97  const double *lb, const double *ub,
98  double *x, double *minf, bopt_params parameters)
99 {
100  vectord result(nDim);
101 
102  vectord lowerBound = bayesopt::utils::array2vector(lb,nDim);
103  vectord upperBound = bayesopt::utils::array2vector(ub,nDim);
104 
105  try
106  {
107  CContinuousModel optimizer(nDim, parameters);
108 
109  optimizer.set_eval_funct(f);
110  optimizer.save_other_data(f_data);
111  optimizer.setBoundingBox(lowerBound,upperBound);
112 
113  optimizer.optimize(result);
114  std::copy(result.begin(), result.end(), x);
115 
116  *minf = optimizer.getValueAtMinimum();
117  }
118  catch (std::bad_alloc& e)
119  {
120  FILE_LOG(logERROR) << e.what();
121  return BAYESOPT_OUT_OF_MEMORY;
122  }
123  catch (std::invalid_argument& e)
124  {
125  FILE_LOG(logERROR) << e.what();
126  return BAYESOPT_INVALID_ARGS;
127  }
128  catch (std::runtime_error& e)
129  {
130  FILE_LOG(logERROR) << e.what();
131  return BAYESOPT_RUNTIME_ERROR;
132  }
133  catch (...)
134  {
135  FILE_LOG(logERROR) << "Unknown error";
136  return BAYESOPT_FAILURE;
137  }
138  return 0; /* everything ok*/
139 };

141 int bayes_optimization_disc(int nDim, eval_func f, void* f_data,
142  double *valid_x, size_t n_points,
143  double *x, double *minf, bopt_params parameters)
144 {
145  vectord result(nDim);
146  vectord input(nDim);
147  vecOfvec xSet;
148 
149  for(size_t i = 0; i<n_points;++i)
150  {
151  for(int j = 0; j<nDim; ++j)
152  {
153  input(j) = valid_x[i*nDim+j];
154  }
155  xSet.push_back(input);
156  }
157 
158  if(parameters.n_init_samples > n_points)
159  {
160  parameters.n_init_samples = n_points;
161  parameters.n_iterations = 0;
162  }
163 
164  try
165  {
166  CDiscreteModel optimizer(xSet,parameters);
167 
168  optimizer.set_eval_funct(f);
169  optimizer.save_other_data(f_data);
170  optimizer.optimize(result);
171 
172  std::copy(result.begin(), result.end(), x);
173 
174  *minf = optimizer.getValueAtMinimum();
175  }
176  catch (std::bad_alloc& e)
177  {
178  FILE_LOG(logERROR) << e.what();
179  return BAYESOPT_OUT_OF_MEMORY;
180  }
181  catch (std::invalid_argument& e)
182  {

185  }
