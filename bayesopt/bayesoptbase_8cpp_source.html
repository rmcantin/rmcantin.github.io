74  void BayesOptBase::stepOptimization()
75  {
76  // Find what is the next point.
77  vectord xNext = nextPoint();
78  double yNext = evaluateSampleInternal(xNext);
79 
80  // If we are stuck in the same point for several iterations, try a random jump!
81  if (mParameters.force_jump)
82  {
83  if (std::pow(mYPrev - yNext,2) < mParameters.noise)
84  {
85  mCounterStuck++;
86  FILE_LOG(logDEBUG) << "Stuck for "<< mCounterStuck << " steps";
87  }
88  else
89  {
90  mCounterStuck = 0;
91  }
92  mYPrev = yNext;
93 
94  if (mCounterStuck > mParameters.force_jump)
95  {
96  FILE_LOG(logINFO) << "Forced random query!";
97  xNext = samplePoint();
98  yNext = evaluateSampleInternal(xNext);
99  mCounterStuck = 0;
100  }
101  }
102 
103  mModel->addSample(xNext,yNext);
104 
105  // Update surrogate model
106  bool retrain = ((mParameters.n_iter_relearn > 0) &&
107  ((mCurrentIter + 1) % mParameters.n_iter_relearn == 0));
108 
109  if (retrain) // Full update
110  {
111  mModel->updateHyperParameters();
112  mModel->fitSurrogateModel();
113  }
114  else // Incremental update
115  {
116  mModel->updateSurrogateModel();
117  }
118  plotStepData(mCurrentIter,xNext,yNext);
119  mModel->updateCriteria(xNext);
120  mCurrentIter++;
121  }

160  vectord BayesOptBase::nextPoint()
161  {
162 
163  //Epsilon-Greedy exploration (see Bull 2011)
164  if ((mParameters.epsilon > 0.0) && (mParameters.epsilon < 1.0))
165  {
166  randFloat drawSample(mEngine,realUniformDist(0,1));
167  double result = drawSample();
168  FILE_LOG(logINFO) << "Trying random jump with prob:" << result;
169  if (mParameters.epsilon > result)
170  {
171  FILE_LOG(logINFO) << "Epsilon-greedy random query!";
172  return samplePoint();
173  }
174  }
175 
176  vectord Xnext(mDims);
177 
178  // GP-Hedge and related algorithms
179  if (mModel->criteriaRequiresComparison())
180  {
181  bool changed = true;
182 
183  mModel->setFirstCriterium();
184  while (changed)
185  {
186  findOptimal(Xnext);
187  changed = mModel->setNextCriterium(Xnext);
188  }
189  std::string name = mModel->getBestCriteria(Xnext);
190  FILE_LOG(logINFO) << name << " was selected.";
191  }
192  else // Standard "Bayesian optimization"
193  {
194  FILE_LOG(logDEBUG) << "------ Optimizing criteria ------";
195  findOptimal(Xnext);
196  }
197  return Xnext;
198  }
