49  void PosteriorFixed::setCriteria(randEngine& eng)

51  CriteriaFactory mCFactory;
52 
53  mCrit.reset(mCFactory.create(mParameters.crit_name,mGP.get()));
54  mCrit->setRandomEngine(eng);
55 
56  if (mCrit->nParameters() == mParameters.n_crit_params)
57  {
58  mCrit->setParameters(utils::array2vector(mParameters.crit_params,
59  mParameters.n_crit_params));
60  }
61  else // If the number of paramerters is different, use default.
62  {
63  if (mParameters.n_crit_params != 0)
64  {
65  FILE_LOG(logERROR) << "Expected " << mCrit->nParameters()
66  << " parameters. Got "

69  FILE_LOG(logINFO) << "Usign default parameters for criteria.";
