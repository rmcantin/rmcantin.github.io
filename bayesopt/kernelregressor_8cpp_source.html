53  {

55  vectord newK = computeCrossCorrelation(lastX);
56  newK(newK.size()-1) += mRegularizer; // We add it to the last element

62  void KernelRegressor::computeCholeskyCorrelation()

67  // const matrixd K = computeCorrMatrix();
