50 learning_type str2learn(const char* name)
51 {
52  if (!strcmp(name,"L_FIXED") || !strcmp(name,"fixed")) return L_FIXED;
53  else if (!strcmp(name,"L_EMPIRICAL") || !strcmp(name,"empirical")) return L_EMPIRICAL;
54  else if (!strcmp(name,"L_DISCRETE") || !strcmp(name,"discrete")) return L_DISCRETE;
55  else if (!strcmp(name,"L_MCMC") || !strcmp(name,"mcmc")) return L_MCMC;
56  else return L_ERROR;
57 }
