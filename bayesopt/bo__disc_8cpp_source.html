66 int main(int nargs, char *args[])
67 {
68  // Common configuration
69  // See parameters.h for the available options
70  // This is just an example of what can be done.
71  bopt_params par = initialize_parameters_to_default();
72 
73  set_surrogate(&par,"sStudentTProcessJef");
74  par.noise = 1e-10;
75  par.n_iterations = 70; // Number of iterations
76  par.n_init_samples = 20;
77  /*******************************************/
78 
79  const size_t nPoints = 1000; // Number of discrete points
80  const size_t n = 6; // Number of dimensions
81 
82  // Let's generate a discrete grid by latin hypercube sampling in the n-dimensional
83  // space.
84  randEngine mtRandom;
85  matrixd xPoints(nPoints,n);
86 
87  bayesopt::utils::lhs(xPoints,mtRandom);
88 
89 
90  // Set of discrete points using C arrays.
91  // Thanks to the quirks of Visual Studio, the following expression is invalid, so we
92  // have to replace it by a literal. WARNING: Make sure to update the size of the array
93  // if the number of points or dimensions change.
94 #ifdef _MSC_VER
95  double xPointsArray[6000];
96 #else
97  const size_t nPinArr = n*nPoints;
98  double xPointsArray[nPinArr];
99 #endif
100 
101  // Set of discrete points using ublas vectors. For the C++ interface.
102  vecOfvec xP;
103 
104 
105  // Lets copy the generated points to the specific structure for each interface.
106  for(size_t i = 0; i<nPoints; ++i)
107  {
108  vectord point = row(xPoints,i);
109  xP.push_back(point);
110  for(size_t j=0; j<n; ++j)
111  {
112  xPointsArray[i*n+j] = point(j);
113  }
114  }
115 
116 
117  /*******************************************/
118  // Run C++ interface
119  std::cout << "Running C++ interface" << std::endl;
120  ExampleDisc opt(xP,par);
121  vectord result(n);
122  opt.optimize(result);
123 
124 
125  // Run C interface
126  std::cout << "Running C interface" << std::endl;
127  double x[128], fmin[128];
128  bayes_optimization_disc(n, &testFunction, NULL, xPointsArray, nPoints,
129  x, fmin, par);
130 
131 
132  // Find the optimal value by brute force
133  size_t min = 0;
134  double minvalue = opt.evaluateSample(row(xPoints,min));
135  for(size_t i = 1; i<nPoints; ++i)
136  {
137  vectord point = row(xPoints,i);
138  if (opt.evaluateSample(point) < minvalue)
139  {
140  min = i;
141  minvalue = opt.evaluateSample(row(xPoints,min));
142  std::cout << i << "," << minvalue << std::endl;
143  }
144  }
145 
146  std::cout << "Final result C++: " << result << " | Value:" << opt.evaluateSample(result) << std::endl;
147  std::cout << "Final result C: ["<< n << "](";
148  for (int i = 0; i < n; i++ )
149  std::cout << x[i] << ", ";
150  std::cout << ")" << " | Value:" << fmin[0] << std::endl;
151  std::cout << "Optimal: " << row(xPoints,min) << " | Value:" << minvalue << std::endl;
152 }
