99  void DiscreteModel::sampleInitialPoints(matrixd& xPoints, vectord& yPoints)
100  {
101 
102  vecOfvec perms = mInputSet;
103 
104  // By using random permutations, we guarantee that
105  // the same point is not selected twice
106  utils::randomPerms(perms,mEngine);
107 
108  // vectord xPoint(mInputSet[0].size());
109  for(size_t i = 0; i < yPoints.size(); i++)
110  {
111  const vectord xP = perms[i];
112  row(xPoints,i) = xP;
113  yPoints(i) = evaluateSample(xP);
114  }
115  }

118  void DiscreteModel::findOptimal(vectord &xOpt)
119  {
120  xOpt = *mInputSet.begin();
121  double min = evaluateCriteria(xOpt);
122 
123  for(vecOfvec::iterator it = mInputSet.begin();
124  it != mInputSet.end(); ++it)
125  {
126  double current = evaluateCriteria(*it);
127  if (current < min)
128  {
129  xOpt = *it;
130  min = current;
131  }
132  }
133  }
